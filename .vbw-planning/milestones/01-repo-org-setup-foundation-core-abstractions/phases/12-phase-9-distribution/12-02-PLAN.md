# Phase 12 / Plan 02: `swt update` CLI command + version checking

## Why second

`swt update` is the user-facing arm of REQ-17. Plan 01 makes the package publishable; Plan 02 lets users discover when a new version is available without leaving the CLI. The command is read-only — it does NOT auto-install (that requires shell privilege escalation in many environments) — it just surfaces the install command for the user to run.

packages/cli/src/main.ts          # register update command

```

## Tasks

### T1 — npm registry helper
Author `packages/cli/src/lib/npm-registry.ts`:
```typescript
import { mkdirSync, readFileSync, writeFileSync, existsSync } from 'node:fs';
import { homedir } from 'node:os';
import { join } from 'node:path';

const DEFAULT_REGISTRY = 'https://registry.npmjs.org';
const CACHE_TTL_MS = 24 * 60 * 60 * 1000;

export interface RegistryResult {
  current: string;
  latest: string;
  status: 'up-to-date' | 'outdated' | 'unreachable';
  error?: string;
  cached?: boolean;
}

export interface QueryOptions {
  registry?: string;
  cachePath?: string;
  noCache?: boolean;
  fetchImpl?: typeof fetch;
}

export async function queryLatestVersion(
  packageName: string,
  current: string,
  opts: QueryOptions = {}
): Promise<RegistryResult> {
  const cachePath = opts.cachePath ?? join(homedir(), '.swt', 'update-cache.json');
  const registry = opts.registry ?? DEFAULT_REGISTRY;

  if (!opts.noCache && existsSync(cachePath)) {
    try {
      const cache = JSON.parse(readFileSync(cachePath, 'utf8'));
      if (cache[packageName] && Date.now() - cache[packageName].at < CACHE_TTL_MS) {
        return { ...cache[packageName].value, current, cached: true };
      }
    } catch { /* corrupted cache — ignore */ }
  }

  const fetchFn = opts.fetchImpl ?? fetch;
  const url = `${registry}/${encodeURIComponent(packageName)}/latest`;

  try {
    const res = await fetchFn(url);
    if (!res.ok) {
      return { current, latest: current, status: 'unreachable', error: `HTTP ${res.status}` };
    }
    const json = await res.json() as { version: string };
    const latest = json.version;
    const status: RegistryResult['status'] = latest === current ? 'up-to-date' : 'outdated';
    const result: RegistryResult = { current, latest, status };

    try {
      const dir = join(homedir(), '.swt');
      mkdirSync(dir, { recursive: true });
      const cache = existsSync(cachePath) ? JSON.parse(readFileSync(cachePath, 'utf8')) : {};
      cache[packageName] = { at: Date.now(), value: { current, latest, status } };
      writeFileSync(cachePath, JSON.stringify(cache, null, 2) + '\n');
    } catch { /* cache write failure — proceed with result */ }

    return result;
  } catch (err) {
    return {
      current,
      latest: current,
      status: 'unreachable',
      error: err instanceof Error ? err.message : String(err),
    };
  }
}
```

### T2 — update command
Author `packages/cli/src/commands/update.ts`:
```typescript
import packageJson from '../../package.json' with { type: 'json' };
import { queryLatestVersion } from '../lib/npm-registry.js';

export interface UpdateOptions {
  json?: boolean;
  strict?: boolean;
  registry?: string;
  noCache?: boolean;
}

export async function updateCommand(opts: UpdateOptions): Promise<number> {
  const result = await queryLatestVersion('@swt-labs/cli', packageJson.version, {
    registry: opts.registry,
    noCache: opts.noCache,
  });

  if (opts.json) {
    const payload: any = {
      status: result.status,
      current: result.current,
      latest: result.latest,
    };
    if (result.status === 'outdated') {
      payload.upgrade_commands = [
        'npm install -g @swt-labs/cli@latest',
        'pnpm add -g @swt-labs/cli@latest',
        'bun add -g @swt-labs/cli@latest',
      ];
    }
    if (result.error) payload.error = result.error;
    process.stdout.write(JSON.stringify(payload, null, 2) + '\n');
  } else {
    if (result.status === 'up-to-date') {
      console.log(`✓ swt is up-to-date (v${result.current})`);
    } else if (result.status === 'outdated') {
      console.log(`↑ Update available: v${result.current} → v${result.latest}\n`);
      console.log('Upgrade with one of:');
      console.log(`  npm install -g @swt-labs/cli@latest`);
      console.log(`  pnpm add -g @swt-labs/cli@latest`);
      console.log(`  bun add -g @swt-labs/cli@latest`);
    } else {
      console.warn(`⚠ Could not check for updates: ${result.error ?? 'unknown'}`);
    }
  }

  if (opts.strict && result.status === 'unreachable') return 1;
  return 0;
}
```

## `swt update`

Check the npm registry for a newer published version.

```bash
swt update [--json] [--strict] [--registry <url>] [--no-cache]
```

| Flag | Type | Default | Description |
|------|------|---------|-------------|
| `--json` | boolean | false | Machine-readable output |
| `--strict` | boolean | false | Exit 1 if registry unreachable (default: warn-only) |
| `--registry` | string | https://registry.npmjs.org | Override registry URL |
| `--no-cache` | boolean | false | Bypass the 24h cache at `~/.swt/update-cache.json` |

Examples:

\`\`\`bash
# Default — human-readable
swt update

# CI usage with strict mode
swt update --json --strict | jq .latest
\`\`\`
