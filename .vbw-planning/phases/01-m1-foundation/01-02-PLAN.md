   ```ts
   export interface MeterRecord {
     timestamp: string;
     milestone: string;
     phase: string;
     task_id: string;
     role: string;
     tier: string;
     provider: string;
     model: string;
     turn: number;
     input: number;
     output: number;
     cacheRead: number;
     cacheWrite: number;
     cost_usd: number;
   }

   export function createTokenMeter(opts?: { persist?: boolean; persistPath?: string }): TokenMeter {
     const records: MeterRecord[] = [];
     const listeners: Array<(event: MeterUpdate) => void> = [];
     return {
       record(r: Omit<MeterRecord, 'cost_usd'>, costUsd: number) {
         const full = { ...r, cost_usd: costUsd };
         records.push(full);
         listeners.forEach(l => l({ type: 'METER_UPDATED', record: full }));
         if (opts?.persist) appendJsonl(opts.persistPath, full);
       },
       snapshot() {
         return aggregateByDimensions(records); // by task / phase / milestone / provider
       },
       subscribe(listener) {
         listeners.push(listener);
         return () => { listeners.splice(listeners.indexOf(listener), 1); };
       },
     };
   }
   ```

                                       overrides?: { roleTier?: RoleTierMap; tierModel?: DefaultTierMap }): string {
     const tier = resolveTierForRole(role, overrides?.roleTier);
     const tierMap = overrides?.tierModel?.[provider] ?? (defaults as DefaultTierMap)[provider];
     if (!tierMap) throw new Error(`Provider not in tier map: ${provider}`);
