### T4 — config-doc drift check
Author `packages/cli/test/config-doc-drift.test.ts`:

```typescript
import { describe, expect, it } from 'vitest';
import { readFileSync } from 'node:fs';
import { join } from 'node:path';
import { ConfigSchema } from '@swt-labs/core';

const DOCS_PATH = join(__dirname, '..', '..', '..', 'docs', 'reference', 'config.mdx');

describe('config-doc drift', () => {
  it('every Zod schema key appears in docs/reference/config.mdx', () => {
    const docs = readFileSync(DOCS_PATH, 'utf8');
    const schemaKeys = Object.keys(ConfigSchema.shape);
    const missing: string[] = [];
    for (const key of schemaKeys) {
      // Match `### \`key\`` or `### key` heading
      const heading = new RegExp(`### \`?${key}\`?`, 'i');
      if (!heading.test(docs)) missing.push(key);
    }
    expect(missing, `keys missing from docs: ${missing.join(', ')}`).toEqual([]);
  });
});
```

Note: `ConfigSchema.shape` access depends on Zod version. If `.shape` is private, use the workaround `(ConfigSchema as any)._def.shape()`.
